377bool MainWindow::LoadModel(const QString& fileName)
378{
379 QFile file(fileName);
380
381 if (!file.open(QIODevice::ReadOnly))
382 {
383 QMessageBox::critical(
384 this,
385 tr("Load Error"),
386 tr("Could not open file for reading:\n%1\n\nError: %2")
387 .arg(fileName)
388 .arg(file.errorString())
389 );
390 return false;
391 }
392
393 // Read and parse JSON
394 QByteArray fileData = file.readAll();
395 file.close();
396
397 QJsonDocument jsonDoc = QJsonDocument::fromJson(fileData);
398 if (jsonDoc.isNull())
399 {
400 QMessageBox::critical(
401 this,
402 tr("Load Error"),
403 tr("Invalid JSON format in file:\n%1")
404 .arg(fileName)
405 );
406 return false;
407 }
408
409 QJsonObject jsonObject = jsonDoc.object();
410
411 // Set working directory
412 QFileInfo fileInfo(fileName);
413 conductor->SetWorkingFolder(fileInfo.absolutePath());
414
415 // Load data
416 file.open(QIODevice::ReadOnly);
417 Data()->ReadFromFile(&file);
418 file.close();
419
420 // Load results
421 QJsonObject resultsJson = jsonObject["Results"].toObject();
422 resultsViewModel->clear();
423 resultsViewModel->setHorizontalHeaderLabels(QStringList() << "Results");
424
425 for (const QString& key : resultsJson.keys())
426 {
427 ResultSetItem* resultSet = new ResultSetItem(key);
428 resultSet->setToolTip(key);
429 resultSet->result = new Results();
430 resultSet->result->ReadFromJson(resultsJson.value(key).toObject());
431
432 // Process MLR results
433 for (auto it = resultSet->result->begin(); it != resultSet->result->end(); ++it)
434 {
435 if (it->second.Type() == result_type::mlrset)
436 {
437 QStringList parts = QString::fromStdString(it->first).split("OM & Size MLR for ");
438 if (parts.size() > 1)
439 {
440 std::string source = parts[1].toStdString();
441 if (Data()->count(source) > 0)
442 {
443 Data()->at(source).SetRegressionModels(
444 static_cast<MultipleLinearRegressionSet*>(it->second.Result())
445 );
446 Data()->SetOMandSizeConstituents(
447 Data()->at(source).GetRegressionModels()->begin()->second.GetIndependentVariableNames()
448 );
449 }
450 }
451 }
452 }
453
454 resultsViewModel->appendRow(resultSet);
455 }
456
457 // Finalize data
458 dataCollection.PopulateElementDistributions();
459 dataCollection.AssignAllDistributions();
460 InitiateTables();
461
462 // Update UI state
463 ProjectFileName = fileName;
464 setWindowTitle("SedSat3: " + ProjectFileName);
465
466 return true;
467}

535bool MainWindow::ReadExcel(const QString& filename)
536{
537 // Load Excel file
538 Document xlsxR(filename);
539 if (!xlsxR.load())
540 {
541 QMessageBox::critical(
542 this,
543 tr("Excel Load Error"),
544 tr("Failed to load Excel file:\n%1").arg(filename)
545 );
546 return false;
547 }
548
549 qDebug() << "[debug] Successfully loaded xlsx file.";
550 QStringList allSheetNames = xlsxR.sheetNames();
551
552 // Clear existing data
553 dataCollection.Clear();
554
555 // Let user select sheets to import
556 QStringList* sheetsToInclude = nullptr;
557 auto chooseSheetsDlg = std::make_unique<DialogChooseExcelSheets>(this, &sheetsToInclude);
558
559 for (const QString& sheetName : allSheetNames)
560 {
561 chooseSheetsDlg->AddItem(sheetName);
562 qDebug() << sheetName;
563 }
564
565 chooseSheetsDlg->exec();
566
567 if (!sheetsToInclude)
568 {
569 return false; // User cancelled
570 }
571
572 // Filter selected sheets
573 QStringList selectedSheets;
574 for (const QString& sheetName : allSheetNames)
575 {
576 if (sheetsToInclude->contains(sheetName))
577 {
578 selectedSheets << sheetName;
579 }
580 }
581 delete sheetsToInclude;
582
583 if (selectedSheets.isEmpty())
584 {
585 return false; // No sheets selected
586 }
587
588 // Let user indicate which sheet is the sink
589 auto indicateSheetDlg = std::make_unique<IndicateSheetsDialog>(this);
590 indicateSheetDlg->Populate_Table(selectedSheets);
591 indicateSheetDlg->exec();
592
593 qDebug() << "Sink Sheet is:" << sinkSheet;
594
595 // Validate element names are consistent across all sheets
596 QList<QStringList> elementNamesBySheet;
597
598 for (int sheetIdx = 0; sheetIdx < selectedSheets.count(); ++sheetIdx)
599 {
600 xlsxR.selectSheet(selectedSheets[sheetIdx]);
601 QStringList elementNames;
602
603 // Read element names from row 1, starting at column 2
604 int col = 2;
605 while (true)
606 {
607 auto cell = xlsxR.cellAt(1, col); // Returns shared_ptr
608 if (!cell)
609 break;
610
611 QString elemName = cell->readValue().toString().trimmed();
612 if (elemName.isEmpty())
613 break;
614
615 elementNames.append(elemName);
616 qDebug() << elemName;
617 col++;
618 }
619
620 // Verify element names match across sheets
621 if (sheetIdx > 0 && elementNames != elementNamesBySheet.last())
622 {
623 QMessageBox::warning(
624 this,
625 tr("Element Mismatch"),
626 tr("Element names and their order must be identical in all sheets.\n\n"
627 "Mismatch found between:\n '%1'\n '%2'")
628 .arg(selectedSheets[sheetIdx - 1])
629 .arg(selectedSheets[sheetIdx])
630 );
631
632 WriteMessageOnScreen(
633 QString("Element mismatch between sheets '%1' and '%2'")
634 .arg(selectedSheets[sheetIdx - 1])
635 .arg(selectedSheets[sheetIdx]),
636 Qt::red
637 );
638 return false;
639 }
640
641 elementNamesBySheet.append(elementNames);
642 }
643
644 // Read sample data from each sheet
645 for (int sheetIdx = 0; sheetIdx < selectedSheets.count(); ++sheetIdx)
646 {
647 xlsxR.selectSheet(selectedSheets[sheetIdx]);
648
649 Elemental_Profile_Set* profileSet = dataCollection.AppendSampleSet(
650 selectedSheets[sheetIdx].toStdString()
651 );
652
653 // Read samples starting from row 2
654 int row = 2;
655 while (true)
656 {
657 auto sampleCell = xlsxR.cellAt(row, 1);
658 if (!sampleCell || sampleCell->readValue().toString().isEmpty())
659 break;
660
661 QString sampleName = sampleCell->readValue().toString();
662 Elemental_Profile elementalProfile;
663
664 // Read element values for this sample
665 for (int col = 0; col < elementNamesBySheet[0].count(); ++col)
666 {
667 auto dataCell = xlsxR.cellAt(row, col + 2);
668
669 if (!dataCell)
670 {
671 QMessageBox::warning(
672 this,
673 tr("Empty Cell"),
674 tr("Empty cell found in:\n"
675 "Sheet: %1\n"
676 "Row: %2\n"
677 "Column: %3")
678 .arg(selectedSheets[sheetIdx])
679 .arg(row)
680 .arg(col + 2)
681 );
682 dataCollection.Clear();
683 return false;
684 }
685
686 bool isNumber = false;
687 double value = dataCell->readValue().toDouble(&isNumber);
688
689 if (!isNumber)
690 {
691 QMessageBox::warning(
692 this,
693 tr("Invalid Data"),
694 tr("Non-numerical value found:\n"
695 "Sheet: %1\n"
696 "Row: %2\n"
697 "Column: %3\n"
698 "Value: '%4'")
699 .arg(selectedSheets[sheetIdx])
700 .arg(row)
701 .arg(col + 2)
702 .arg(dataCell->readValue().toString())
703 );
704 dataCollection.Clear();
705 return false;
706 }
707
708 elementalProfile.AppendElement(
709 elementNamesBySheet[0][col].toStdString(),
710 value
711 );
712 }
713
714 profileSet->AppendProfile(sampleName.toStdString(), elementalProfile);
715 row++;
716 }
717 }
718
719 // Finalize data collection
720 dataCollection.PopulateElementInformation();
721 dataCollection.PopulateElementDistributions();
722 dataCollection.AssignAllDistributions();
723
724 qDebug() << "Reading element information done!";
725
726 return true;
727}

1339void MainWindow::addToRecentFiles(const QString& fileName, bool addToFile)
1340{
1341 if (fileName.trimmed().isEmpty())
1342 {
1343 return;
1344 }
1345
1346 QString trimmedFileName = fileName.trimmed();
1347 bool rewriteFile = false;
1348
1349 // If file already exists in list, move it to the end (most recent)
1350 if (recentFiles.contains(trimmedFileName))
1351 {
1352 int existingIndex = recentFiles.indexOf(trimmedFileName);
1353
1354 // Only move if not already at the end
1355 if (existingIndex != recentFiles.count() - 1)
1356 {
1357 // Remove from menu
1358 QList<QAction*> actions = ui->menuRecent->actions();
1359 int actionIndex = recentFiles.size() - 1 - existingIndex;
1360 if (actionIndex >= 0 && actionIndex < actions.size())
1361 {
1362 ui->menuRecent->removeAction(actions[actionIndex]);
1363 }
1364
1365 // Remove from list
1366 recentFiles.removeOne(trimmedFileName);
1367 addToFile = false;
1368 rewriteFile = true;
1369 }
1370 else
1371 {
1372 return; // Already most recent, nothing to do
1373 }
1374 }
1375
1376 // Enforce maximum number of recent files
1377 if (recentFiles.size() >= max_num_recent_files)
1378 {
1379 // Remove oldest file (first in list)
1380 recentFiles.removeFirst();
1381
1382 // Remove oldest action (last in menu)
1383 QList<QAction*> actions = ui->menuRecent->actions();
1384 if (!actions.isEmpty())
1385 {
1386 ui->menuRecent->removeAction(actions.last());
1387 }
1388
1389 rewriteFile = true;
1390 }
1391
1392 // Add to list
1393 recentFiles.append(trimmedFileName);
1394
1395 // Add to menu (at the top)
1396 QAction* fileAction = new QAction(trimmedFileName, this);
1397
1398 QList<QAction*> actions = ui->menuRecent->actions();
1399 if (!actions.isEmpty())
1400 {
1401 ui->menuRecent->insertAction(actions.first(), fileAction);
1402 }
1403 else
1404 {
1405 ui->menuRecent->addAction(fileAction);
1406 }
1407
1408 connect(fileAction, &QAction::triggered, this, &MainWindow::on_actionRecent_triggered);
1409
1410 // Write to file if needed
1411 if (addToFile)
1412 {
1413 QString recentFilePath = QStandardPaths::writableLocation(QStandardPaths::DocumentsLocation)
1414 + "/" + RECENT;
1415 CreateFileIfDoesNotExist(recentFilePath);
1416
1417 QFile file(recentFilePath);
1418 if (file.open(QIODevice::Append | QIODevice::Text))
1419 {
1420 QTextStream out(&file);
1421 out << trimmedFileName << "\n";
1422 file.close();
1423 }
1424 }
1425
1426 if (rewriteFile)
1427 {
1428 writeRecentFilesList();
1429 }
1430}
